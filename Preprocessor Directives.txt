2. main preprocessor directives

	1. #define -> macro definition

	ex:- 	#define PI 3.14
	     	#define SQR(x) ((x)*(x))

	2. #include -> file inclusion

	ex:-	#include<stdio.h>  //< > system headers

	 	#include<string.h>  // " " user defined headers

	3. #undef -> remove macro
		#define pI 3.14
		#undef PI

****************File Pointer********************

FILE *fp;

file is a structure defined in stdio.h

opening file:

fp = fopen("data.txt", "r");

"r" read only			"r+" read + write

"w" write only			"w+" read + write + truncate

"a" append			"a+" read + append

"rb"  binary read		"wb" binary write

"ab"  binary append


FILE * is a pointer that stores the address of the opened file and is returned by fopen().

fopen() open file    		fseek()  move pointer

fprintf() write text		fwrite() binary write	

fgetc() read character		fread() binary read

fclose close file 		remind(fp) moves file pointer to beginning


fseek( fp, 0, SEEK_END)   moves to end	//SEEK_END  end

fseek(fp, 0, SEEK_SET)  moves to beginning  //SEEK_SET  start

fseek(fp, -1, SEEK_CUR) moves back 1 byte only	//SEEK_CUR   current

fclose() fopen() reopens file, not equivalent
